Solving Square Root


	1, double number: x
	2, start with a guess: g
	3, calculate: x / g
	4, check if x/g close enough to g
	   if not
	      make a new guess: g and repeat steps 3, 4
	   if yes
		  g is your answer 
	5, return g

sqrt implementation


	public static double sqrt(double x) {
		double g = x/2.0;
		double b = x/g;
		while(Math.abs(g-b)>0.0001) {
			g = (g+b)/2.0;
			b = x/g;
			// Debugging output
			System.out.println("b: "+b+", g="+g);
		}
		return g;
	}
